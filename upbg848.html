实现样式修改的核心逻辑 

// 获取原始元素
const originalContent = document.getElementById('originalContent');

// 复制内容用于显示
document.getElementById('originalDisplay').innerHTML = originalContent.innerHTML;
document.getElementById('modifiedDisplay').innerHTML = originalContent.innerHTML;

// 修改样式函数
function changeStyles() {
    const content = document.getElementById('modifiedDisplay');
    
    // 修改样式属性
    content.style.fontFamily = "'Georgia', serif";
    content.style.fontSize = "18px";
    content.style.lineHeight = "1.8";
    content.style.color = "#2c3e50";
    content.style.backgroundColor = "#f9f7f1";
    content.style.padding = "25px";
    content.style.borderRadius = "10px";
    content.style.boxShadow = "0 5px 15px rgba(0,0,0,0.1)";
    content.style.borderLeft = "5px solid #e74c3c";
    
    // 修改特定文本样式
    const text = content.innerHTML;
    content.innerHTML = text.replace(/Singapore/g, 'Singapore');
}

// 高亮重要内容
function highlightImportant() {
    const content = document.getElementById('modifiedDisplay');
    let text = content.innerHTML;
    
    // 高亮关键信息
    text = text.replace(/no sun protection measures/g, 'no sun protection measures');
    text = text.replace(/no hats/g, 'no hats');
    text = text.replace(/no UPF clothing/g, 'no UPF clothing');
    text = text.replace(/not even sunscreen/g, 'not even sunscreen');
    
    content.innerHTML = text;
}

// 更改布局
function changeLayout() {
    const content = document.getElementById('modifiedDisplay');
    content.style.columns = "2";
    content.style.columnGap = "30px";
    content.style.columnRule = "2px dashed #3498db";
}

// 重置样式
function resetStyles() {
    const content = document.getElementById('modifiedDisplay');
    content.style = "";
    content.innerHTML = document.getElementById('originalContent').innerHTML;
}

// 添加事件监听器
document.getElementById('changeStyle').addEventListener('click', changeStyles);
document.getElementById('resetStyle').addEventListener('click', resetStyles);
document.getElementById('highlightText').addEventListener('click', highlightImportant);
document.getElementById('changeLayout').addEventListener('click', changeLayout);

提示： 这个示例展示了如何使用JavaScript获取HTML元素并动态修改其样式。在实际项目中，这种方法可以用于创建交互式用户界面、动态主题切换或响应式内容调整。
